import random
import time

def wake_up():
    print("Woke up... or maybe not, just 5 more minutes")

    if random.choice([True, False]):
        print("Got up successfully!")
    else:
        print("Overslept again üòÖ")

    if age < 18:
        print("Going to school... or pretending to study")
    else:
        print("Going to work, pretending to be an adult... almost nailed it üòé")

def eat_meal():
    meals = ["ham sandwich", "pizza", "something edible from the kitchen", "porridge"]
    choice = random.choice(meals)
    print(f"Today's meal: {choice}")

    actions = ["learning something new", "playing games", "being lazy", "scrolling memes"]
    print(f"Doing... {random.choice(actions)}")

    print("Sleep time... hoping for deep sleep")
    time.sleep(1)
    print("Sleeping üò¥")

    wake_up()
    eat_meal()
    go_to_school_or_work(age)

    eat_meal()
